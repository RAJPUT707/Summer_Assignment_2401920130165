class Solution {
    public int strStr(String haystack, String needle) {
        int l1 = haystack.length();
        int l2 = needle.length();
        
        if(l2 == 0) return 0;  

        for(int j = 0; j + l2 <= l1; j++) {
            String a = haystack.substring(j, j + l2);
            if(a.equals(needle)) {
                return j;
            }
        }

        return -1;  
    }
}
