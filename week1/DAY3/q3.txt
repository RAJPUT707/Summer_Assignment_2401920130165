class Solution {
    public int maxArea(int[] height) {
        int i = 0, j = height.length - 1;
        int ans = 0;

        while (i < j) {
            int small = Math.min(height[i], height[j]);
            int area = small * (j - i);
            ans = Math.max(ans, area);

            // Move the pointer with smaller height
            if (height[i] < height[j]) {
                i++;
            } else {
                j--;
            }
        }

        return ans;
    }
}
