class Solution {
    public void moveZeroes(int[] nums) {
        int l = nums.length;
        int zero = 0;
        for(int i = 0;i<l;i++)
        {
if(nums[i]==0)
{
    zero++;
}
        }
        int c = 0;
        for(int i = 0;i<l;i++)
        {
            if(nums[i]!=0)
            {
                nums[c++]=nums[i];
            }}
            for(int i = c;i<l;i++)
            {
                nums[i]=0;
            }
        }
    }
