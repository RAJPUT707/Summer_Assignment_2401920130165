class Solution {
    public int[][] matrixReshape(int[][] mat, int r, int c) {
      int ans[][] = new int[r][c];
      int x =0;
      int y =0;
      if(r*c!=(mat.length*mat[0].length))
      {
        return mat;
      }
      for(int i=0;i<mat.length;i++)
      {
        for(int j=0;j<mat[0].length;j++)
        {
            if(y==c)
            {
                y=0;
                x++;
            }
            ans[x][y]=mat[i][j];
            y++;
        }
      }
      return ans;  
    }
}
